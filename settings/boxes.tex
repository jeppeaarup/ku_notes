% --- Color boxes ---
\usepackage[theorems, many]{tcolorbox}

\ifthenelse{\boolean{usechapters}}{
    \newcommand{\counterparent}{chapter}
}{
    \newcommand{\counterparent}{section}
}

% Define box styles ---
\tcbset{
    framedboxstyle/.style={
        fonttitle=\bfseries,
        coltitle=black,
        toptitle=3mm,
        bottomtitle=-0.5mm,
        separator sign = \quad,
        colbacktitle=lightgrey,
        colback=lightgrey,
        boxrule=1pt,
        titlerule=0pt,
        sharp corners,
        enhanced jigsaw,
    },
    leftruleboxstyle/.style={
        fonttitle=\bfseries,
        coltitle=black,
        toptitle=3mm,
        bottomtitle=-0.5mm,
        separator sign = \quad,
        colbacktitle=lightgrey,
        colback=lightgrey,
        boxrule=0pt,
        titlerule=0pt,
        frame empty,
        sharp corners,
        enhanced jigsaw,
    },
    leftrule color/.style={
        borderline west={3pt}{0pt}{#1}
    }
}

% Definition environment
\newtcbtheorem[number within=\counterparent]{definition}{Definition}{
    framedboxstyle,
    colframe=darkgreen,
}{def}

% Theorem environment
\newtcbtheorem[number within=\counterparent]{theorem}{Theorem}{
    framedboxstyle,
    colframe=darkblue,
}{thm}

% Method environment
\newtcbtheorem[number within=\counterparent]{method}{Method}{
    framedboxstyle,
    colframe=black,
}{mth}

% Example environment
\newtcbtheorem[number within=\counterparent]{example}{Example}{
    leftruleboxstyle,
    leftrule color=brightorange
}{exa}

% Box environment
\newtcbtheorem[number within=\counterparent]{highlight}{Box}{
    leftruleboxstyle,
    leftrule color=darkgreen
}{box}

% Proof environment
\newtcbtheorem[number within=\counterparent]{proof}{Proof}{
    leftruleboxstyle,
    leftrule color=black
}{prf}

% Important environment
\newtcolorbox[]{important}{
    colback=whitered,
    colframe=darkred,
    borderline west={4pt}{0pt}{darkred},
    boxrule=0pt,
    sharp corners,
    enhanced jigsaw,
}
