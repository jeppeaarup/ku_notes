% --- Page layout ---
\usepackage{geometry} % For customizing page layout
\usepackage{fancyhdr} % Headers, footers, page number etc.

\geometry{a4paper,
hmargin = {20mm, 20mm}, % Right and left margin
vmargin = {30mm, 30mm} % Top and bottom margin
}

% Header and footer setup ---
\renewcommand{\headrulewidth}{0pt}
\fancypagestyle{plain}{
    \fancyhf{}
}

\pagestyle{fancy}
\fancyfoot{}
\fancyhead[R]{\sffamily\thepage}
\ifthenelse{\boolean{usechapters}}{
    \fancyhead[L]{{\bfseries\sffamily Chapter \thechapter \, |} \quad {\transparent{0.6}\sffamily\rightmark}}
}{
    \fancyhead[L]{{\transparent{0.6}\sffamily\leftmark}}
}

% --- Text layout ---
\usepackage{setspace} % For setting line spacing
\usepackage{textpos} % For text positioning
\usepackage{enumitem} % For customizing lists

\onehalfspacing % Set line spacing to 1.5
\setlength{\parindent}{0pt} % No indent on new paragraphs

\setlist{
  itemsep=0pt, % Affects only space between lists and sublists
  parsep=1pt, % Affects spacing between both items and sublists
  topsep=0pt, % Affects top AND bottom of all lists (also between lists and sublists)
  partopsep=5pt % Affects only space between text and list
}

% --- Other layout settings ---
\setcounter{tocdepth}{3} % Set TOC depth
