% Define global box styles
\tcbset{
    framedboxstyle/.style={
    fonttitle=\bfseries, % Title font
    coltitle=black, % Title color
    toptitle=3mm, % Top title padding
    bottomtitle=-0.5mm, % Bottom title padding
    separator sign = \quad, % Seperator between label and title
    colbacktitle=lightgrey, % Title background color
    colback=lightgrey, % Background color
    boxrule=1pt, % Frame width
    titlerule=0pt, % No line under title
    sharp corners, % No rounded corners
    enhanced jigsaw, % Better frame drawing
    },
    leftruleboxstyle/.style={
    fonttitle=\bfseries, % Title font
    coltitle=black, % Title color
    toptitle=3mm, % Top title padding
    bottomtitle=-0.5mm, % Bottom title padding
    separator sign = \quad, % Seperator between label and title
    colbacktitle=lightgrey, % Title background color
    colback=lightgrey, % Background color
    boxrule=0pt, % Frame width
    titlerule=0pt, % No line under title
    frame empty, % No frame
    sharp corners, % No rounded corners
    enhanced jigsaw, % Better frame drawing
    },
    leftrule color/.style={ % For setting color of left rule
        borderline west={3pt}{0pt}{#1}
    }
}
